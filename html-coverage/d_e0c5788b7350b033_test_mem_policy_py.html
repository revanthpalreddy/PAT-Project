92 /* 

93 * This struct allows the dynamic configuration of the allocator funcs

96 */ 

97 typedef struct { 

98 void *(*malloc)(size_t); 

99 void *(*calloc)(size_t, size_t); 

100 void *(*realloc)(void *, size_t); 

101 void (*free)(void *); 

102 } SecretDataAllocatorFuncs; 

103 

104 NPY_NO_EXPORT void * 

105 shift_alloc(void *ctx, size_t sz) { 

106 SecretDataAllocatorFuncs *funcs = (SecretDataAllocatorFuncs *)ctx; 

107 char *real = (char *)funcs->malloc(sz + 64); 

108 if (real == NULL) { 

109 return NULL; 

110 } 

111 snprintf(real, 64, "originally allocated %ld", (unsigned long)sz); 

112 return (void *)(real + 64); 

113 } 

114 NPY_NO_EXPORT void * 

115 shift_zero(void *ctx, size_t sz, size_t cnt) { 

116 SecretDataAllocatorFuncs *funcs = (SecretDataAllocatorFuncs *)ctx; 

117 char *real = (char *)funcs->calloc(sz + 64, cnt); 

118 if (real == NULL) { 

119 return NULL; 

120 } 

121 snprintf(real, 64, "originally allocated %ld via zero", 

122 (unsigned long)sz); 

123 return (void *)(real + 64); 

124 } 

125 NPY_NO_EXPORT void 

126 shift_free(void *ctx, void * p, npy_uintp sz) { 

127 SecretDataAllocatorFuncs *funcs = (SecretDataAllocatorFuncs *)ctx; 

128 if (p == NULL) { 

129 return ; 

130 } 

131 char *real = (char *)p - 64; 

132 if (strncmp(real, "originally allocated", 20) != 0) { 

133 fprintf(stdout, "uh-oh, unmatched shift_free, " 

134 "no appropriate prefix\\n"); 

135 /* Make C runtime crash by calling free on the wrong address */ 

136 funcs->free((char *)p + 10); 

137 /* funcs->free(real); */ 

138 } 

139 else { 

140 npy_uintp i = (npy_uintp)atoi(real +20); 

141 if (i != sz) { 

142 fprintf(stderr, "uh-oh, unmatched shift_free" 

143 "(ptr, %ld) but allocated %ld\\n", sz, i); 

144 /* This happens in some places, only print */ 

145 funcs->free(real); 

146 } 

147 else { 

148 funcs->free(real); 

149 } 

150 } 

151 } 

152 NPY_NO_EXPORT void * 

153 shift_realloc(void *ctx, void * p, npy_uintp sz) { 

154 SecretDataAllocatorFuncs *funcs = (SecretDataAllocatorFuncs *)ctx; 

155 if (p != NULL) { 

156 char *real = (char *)p - 64; 

157 if (strncmp(real, "originally allocated", 20) != 0) { 

158 fprintf(stdout, "uh-oh, unmatched shift_realloc\\n"); 

159 return realloc(p, sz); 

160 } 

161 return (void *)((char *)funcs->realloc(real, sz + 64) + 64); 

162 } 

163 else {

165 if (real == NULL) { 

166 return NULL; 

167 } 

168 snprintf(real, 64, "originally allocated " 

169 "%ld via realloc", (unsigned long)sz); 

170 return (void *)(real + 64); 

171 }
