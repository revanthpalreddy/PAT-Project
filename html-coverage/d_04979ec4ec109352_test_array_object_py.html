23def test_validate_index(): 

24 # The indexing tests in the official array API test suite test that the 

25 # array object correctly handles the subset of indices that are required 

26 # by the spec. But the NumPy array API implementation specifically 

27 # disallows any index not required by the spec, via Array._validate_index. 

28 # This test focuses on testing that non-valid indices are correctly 

29 # rejected. See 

30 # https://data-apis.org/array-api/latest/API_specification/indexing.html 

31 # and the docstring of Array._validate_index for the exact indexing 

32 # behavior that should be allowed. This does not test indices that are 

33 # already invalid in NumPy itself because Array will generally just pass 

34 # such indices directly to the underlying np.ndarray. 

35 

36 a = ones((3, 4)) 

37 

38 # Out of bounds slices are not allowed 

39 assert_raises(IndexError, lambda: a[:4]) 

40 assert_raises(IndexError, lambda: a[:-4]) 

41 assert_raises(IndexError, lambda: a[:3:-1]) 

42 assert_raises(IndexError, lambda: a[:-5:-1]) 

43 assert_raises(IndexError, lambda: a[4:]) 

44 assert_raises(IndexError, lambda: a[-4:]) 

45 assert_raises(IndexError, lambda: a[4::-1]) 

46 assert_raises(IndexError, lambda: a[-4::-1]) 

47 

48 assert_raises(IndexError, lambda: a[...,:5]) 

49 assert_raises(IndexError, lambda: a[...,:-5]) 

50 assert_raises(IndexError, lambda: a[...,:5:-1]) 

51 assert_raises(IndexError, lambda: a[...,:-6:-1]) 

52 assert_raises(IndexError, lambda: a[...,5:]) 

53 assert_raises(IndexError, lambda: a[...,-5:]) 

54 assert_raises(IndexError, lambda: a[...,5::-1]) 

55 assert_raises(IndexError, lambda: a[...,-5::-1]) 

56 

57 # Boolean indices cannot be part of a larger tuple index 

58 assert_raises(IndexError, lambda: a[a[:,0]==1,0]) 

59 assert_raises(IndexError, lambda: a[a[:,0]==1,...]) 

60 assert_raises(IndexError, lambda: a[..., a[0]==1]) 

61 assert_raises(IndexError, lambda: a[[True, True, True]]) 

62 assert_raises(IndexError, lambda: a[(True, True, True),]) 

63 

64 # Integer array indices are not allowed (except for 0-D) 

65 idx = asarray([[0, 1]]) 

66 assert_raises(IndexError, lambda: a[idx]) 

67 assert_raises(IndexError, lambda: a[idx,]) 

68 assert_raises(IndexError, lambda: a[[0, 1]]) 

69 assert_raises(IndexError, lambda: a[(0, 1), (0, 1)]) 

70 assert_raises(IndexError, lambda: a[[0, 1]]) 

71 assert_raises(IndexError, lambda: a[np.array([[0, 1]])]) 

72 

73 # np.newaxis is not allowed 

74 assert_raises(IndexError, lambda: a[None]) 

75 assert_raises(IndexError, lambda: a[None, ...]) 

76 assert_raises(IndexError, lambda: a[..., None]) 

77 

78 # Multiaxis indices must contain exactly as many indices as dimensions 

79 assert_raises(IndexError, lambda: a[()]) 

80 assert_raises(IndexError, lambda: a[0,]) 

81 assert_raises(IndexError, lambda: a[0]) 

82 assert_raises(IndexError, lambda: a[:])
