126class TestDimSpec(util.F2PyTest): 

127 """This test suite tests various expressions that are used as dimension 

128 specifications. 

129 

130 There exists two usage cases where analyzing dimensions 

131 specifications are important. 

132 

133 In the first case, the size of output arrays must be defined based 

134 on the inputs to a Fortran function. Because Fortran supports 

135 arbitrary bases for indexing, for instance, `arr(lower:upper)`, 

136 f2py has to evaluate an expression `upper - lower + 1` where 

137 `lower` and `upper` are arbitrary expressions of input parameters. 

138 The evaluation is performed in C, so f2py has to translate Fortran 

139 expressions to valid C expressions (an alternative approach is 

140 that a developer specifies the corresponding C expressions in a 

141 .pyf file). 

142 

143 In the second case, when user provides an input array with a given 

144 size but some hidden parameters used in dimensions specifications 

145 need to be determined based on the input array size. This is a 

146 harder problem because f2py has to solve the inverse problem: find 

147 a parameter `p` such that `upper(p) - lower(p) + 1` equals to the 

148 size of input array. In the case when this equation cannot be 

149 solved (e.g. because the input array size is wrong), raise an 

150 error before calling the Fortran function (that otherwise would 

151 likely crash Python process when the size of input arrays is 

152 wrong). f2py currently supports this case only when the equation 

153 is linear with respect to unknown parameter. 

154 

155 """ 

156 

157 suffix = ".f90" 

158 

159 code_template = textwrap.dedent(""" 

160 function get_arr_size_{count}(a, n) result (length) 

161 integer, intent(in) :: n 

162 integer, dimension({dimspec}), intent(out) :: a 

163 integer length 

164 length = size(a) 

165 end function 

166 

167 subroutine get_inv_arr_size_{count}(a, n) 

168 integer :: n 

169 ! the value of n is computed in f2py wrapper 

170 !f2py intent(out) n 

171 integer, dimension({dimspec}), intent(in) :: a 

172 if (a({first}).gt.0) then 

173 print*, "a=", a 

174 endif 

175 end subroutine 

176 """) 

177 

178 linear_dimspecs = [ 

179 "n", "2*n", "2:n", "n/2", "5 - n/2", "3*n:20", "n*(n+1):n*(n+5)", 

180 "2*n, n" 

181 ] 

182 nonlinear_dimspecs = ["2*n:3*n*n+2*n"] 

183 all_dimspecs = linear_dimspecs + nonlinear_dimspecs 

184 

185 code = "" 

186 for count, dimspec in enumerate(all_dimspecs): 

187 lst = [(d.split(":")[0] if ":" in d else "1") for d in dimspec.split(',')] 

188 code += code_template.format( 

189 count=count, 

190 dimspec=dimspec, 

191 first=", ".join(lst), 

192 ) 

193 

194 @pytest.mark.parametrize("dimspec", all_dimspecs) 

195 def test_array_size(self, dimspec): 

196 

197 count = self.all_dimspecs.index(dimspec) 

198 get_arr_size = getattr(self.module, f"get_arr_size_{count}") 

199 

200 for n in [1, 2, 3, 4, 5]: 

201 sz, a = get_arr_size(n) 

202 assert a.size == sz 

203 

204 @pytest.mark.parametrize("dimspec", all_dimspecs) 

205 def test_inv_array_size(self, dimspec): 

206 

207 count = self.all_dimspecs.index(dimspec) 

208 get_arr_size = getattr(self.module, f"get_arr_size_{count}") 

209 get_inv_arr_size = getattr(self.module, f"get_inv_arr_size_{count}") 

210 

211 for n in [1, 2, 3, 4, 5]: 

212 sz, a = get_arr_size(n) 

213 if dimspec in self.nonlinear_dimspecs: 

214 # one must specify n as input, the call we'll ensure 

215 # that a and n are compatible: 

216 n1 = get_inv_arr_size(a, n) 

217 else: 

218 # in case of linear dependence, n can be determined 

219 # from the shape of a: 

220 n1 = get_inv_arr_size(a) 

221 # n1 may be different from n (for instance, when `a` size 

222 # is a function of some `n` fraction) but it must produce 

223 # the same sized array 

224 sz1, _ = get_arr_size(n1) 

225 assert sz == sz1, (n, n1, sz, sz1)
